### Security Rules

```javascript
// Firestore security rules for user profiles
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // User profiles - users can only access their own
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // User goals - users can only access their own
    match /user_goals/{goalId} {
      allow read, write: if request.auth != null && 
        request.auth.uid == resource.data.userId;
      allow create: if request.auth != null && 
        request.auth.uid == request.resource.data.userId;
    }
    
    // Health insights - users can only access their own
    match /health_insights/{insightId} {
      allow read, write: if request.auth != null && 
        request.auth.uid == resource.data.userId;
      allow create: if request.auth != null && 
        request.auth.uid == request.resource.data.userId;
    }
  }
}
```

## üíª **Client SDK Examples**

### React/TypeScript Integration

```typescript
// userProfileService.ts
import { getAuth } from 'firebase/auth';
import { doc, getDoc, setDoc, updateDoc, deleteDoc } from 'firebase/firestore';
import { db } from '../config/firebase';

export class UserProfileService {
  private auth = getAuth();

  // Get authenticated user's profile
  async getUserProfile(): Promise<UserProfile | null> {
    const user = this.auth.currentUser;
    if (!user) throw new Error('User not authenticated');

    const profileRef = doc(db, 'users', user.uid);
    const profileSnap = await getDoc(profileRef);
    
    return profileSnap.exists() ? profileSnap.data() as UserProfile : null;
  }

  // Create new user profile
  async createUserProfile(profileData: Partial<UserProfile>): Promise<void> {
    const user = this.auth.currentUser;
    if (!user) throw new Error('User not authenticated');

    const profileRef = doc(db, 'users', user.uid);
    await setDoc(profileRef, {
      ...profileData,
      uid: user.uid,
      email: user.email,
      createdAt: new Date(),
      updatedAt: new Date()
    });
  }

  // Update user profile
  async updateUserProfile(updates: Partial<UserProfile>): Promise<void> {
    const user = this.auth.currentUser;
    if (!user) throw new Error('User not authenticated');

    const profileRef = doc(db, 'users', user.uid);
    await updateDoc(profileRef, {
      ...updates,
      updatedAt: new Date()
    });
  }

  // Delete user profile
  async deleteUserProfile(): Promise<void> {
    const user = this.auth.currentUser;
    if (!user) throw new Error('User not authenticated');

    const profileRef = doc(db, 'users', user.uid);
    await deleteDoc(profileRef);
  }
}
```

### React Hook for Profile Management

```typescript
// useUserProfile.ts
import { useState, useEffect } from 'react';
import { useAuth } from '../contexts/AuthContext';
import { UserProfileService } from '../services/userProfileService';

export const useUserProfile = () => {
  const { currentUser } = useAuth();
  const [profile, setProfile] = useState<UserProfile | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const profileService = new UserProfileService();

  useEffect(() => {
    if (currentUser) {
      loadProfile();
    } else {
      setProfile(null);
      setLoading(false);
    }
  }, [currentUser]);

  const loadProfile = async () => {
    try {
      setLoading(true);
      setError(null);
      const profileData = await profileService.getUserProfile();
      setProfile(profileData);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Failed to load profile');
    } finally {
      setLoading(false);
    }
  };

  const updateProfile = async (updates: Partial<UserProfile>) => {
    try {
      setError(null);
      await profileService.updateUserProfile(updates);
      setProfile(prev => prev ? { ...prev, ...updates } : null);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Failed to update profile');
      throw err;
    }
  };

  const createProfile = async (profileData: Partial<UserProfile>) => {
    try {
      setError(null);
      await profileService.createUserProfile(profileData);
      await loadProfile();
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Failed to create profile');
      throw err;
    }
  };

  return {
    profile,
    loading,
    error,
    updateProfile,
    createProfile,
    refreshProfile: loadProfile
  };
};
```

### Profile Component Example

```typescript
// UserProfile.tsx
import React, { useState } from 'react';
import { useUserProfile } from '../hooks/useUserProfile';

export const UserProfile: React.FC = () => {
  const { profile, loading, error, updateProfile } = useUserProfile();
  const [isEditing, setIsEditing] = useState(false);
  const [editForm, setEditForm] = useState({
    firstName: '',
    lastName: '',
    allergies: [] as string[],
    dietaryRestrictions: [] as string[]
  });

  useEffect(() => {
    if (profile) {
      setEditForm({
        firstName: profile.firstName || '',
        lastName: profile.lastName || '',
        allergies: profile.allergies || [],
        dietaryRestrictions: profile.dietaryRestrictions || []
      });
    }
  }, [profile]);

  const handleSave = async () => {
    try {
      await updateProfile(editForm);
      setIsEditing(false);
    } catch (error) {
      console.error('Failed to update profile:', error);
    }
  };

  if (loading) return <div>Loading profile...</div>;
  if (error) return <div>Error: {error}</div>;
  if (!profile) return <div>No profile found</div>;

  return (
    <div className="max-w-2xl mx-auto p-6">
      <div className="bg-white rounded-lg shadow-md p-6">
        <div className="flex justify-between items-center mb-6">
          <h1 className="text-2xl font-bold">User Profile</h1>
          <button
            onClick={() => setIsEditing(!isEditing)}
            className="px-4 py-2 bg-blue-600 text-white rounded-md hover:bg-blue-700"
          >
            {isEditing ? 'Cancel' : 'Edit'}
          </button>
        </div>

        {isEditing ? (
          <div className="space-y-4">
            <div>
              <label className="block text-sm font-medium text-gray-700">
                First Name
              </label>
              <input
                type="text"
                value={editForm.firstName}
                onChange={(e) => setEditForm({...editForm, firstName: e.target.value})}
                className="mt-1 block w-full rounded-md border-gray-300 shadow-sm"
              />
            </div>

            <div>
              <label className="block text-sm font-medium text-gray-700">
                Last Name
              </label>
              <input
                type="text"
                value={editForm.lastName}
                onChange={(e) => setEditForm({...editForm, lastName: e.target.value})}
                className="mt-1 block w-full rounded-md border-gray-300 shadow-sm"
              />
            </div>

            <div>
              <label className="block text-sm font-medium text-gray-700">
                Allergies
              </label>
              <div className="mt-2 space-y-2">
                {editForm.allergies.map((allergy, index) => (
                  <div key={index} className="flex items-center space-x-2">
                    <input
                      type="text"
                      value={allergy}
                      onChange={(e) => {
                        const newAllergies = [...editForm.allergies];
                        newAllergies[index] = e.target.value;
                        setEditForm({...editForm, allergies: newAllergies});
                      }}
                      className="flex-1 rounded-md border-gray-300 shadow-sm"
                    />
                    <button
                      onClick={() => {
                        const newAllergies = editForm.allergies.filter((_, i) => i !== index);
                        setEditForm({...editForm, allergies: newAllergies});
                      }}
                      className="text-red-600 hover:text-red-800"
                    >
                      Remove
                    </button>
                  </div>
                ))}
                <button
                  onClick={() => setEditForm({...editForm, allergies: [...editForm.allergies, '']})}
                  className="text-blue-600 hover:text-blue-800"
                >
                  Add Allergy
                </button>
              </div>
            </div>

            <div className="flex space-x-4">
              <button
                onClick={handleSave}
                className="px-4 py-2 bg-green-600 text-white rounded-md hover:bg-green-700"
              >
                Save Changes
              </button>
              <button
                onClick={() => setIsEditing(false)}
                className="px-4 py-2 bg-gray-600 text-white rounded-md hover:bg-gray-700"
              >
                Cancel
              </button>
            </div>
          </div>
        ) : (

              <p><strong>Email:</strong> {profile.email}</p>

              <h3 className="text-lg font-medium">Health Information</h3>

              <p><strong>Dietary Restrictions:</strong> {profile.dietaryRestrictions.join(', ') || 'None'}</p>
            </div>
          </div>
        )}
      </div>
    </div>
  );
};
```

```typescript
// Error handling utility
export class UserProfileError extends Error {
  constructor(
    message: string,
    public code: string,
    public statusCode: number = 500
  ) {
    super(message);
    this.name = 'UserProfileError';
  }
}

// Service with proper error handling
export class UserProfileService {
  async getUserProfile(): Promise<UserProfile | null> {
    try {
      const user = this.auth.currentUser;
      if (!user) {
        throw new UserProfileError(
          'User not authenticated',
          'UNAUTHORIZED',
          401
        );
      }

      const profileRef = doc(db, 'users', user.uid);
      const profileSnap = await getDoc(profileRef);
      
      return profileSnap.exists() ? profileSnap.data() as UserProfile : null;
    } catch (error) {
      if (error instanceof UserProfileError) {
        throw error;
      }
      
      console.error('Error getting user profile:', error);
      throw new UserProfileError(
        'Failed to retrieve user profile',
        'FETCH_ERROR',
        500
      );
    }
  }
}
```
